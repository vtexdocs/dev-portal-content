[<i class="fa-brands fa-github"></i> Source code](https://github.com/vtex/faststore/tree/main/packages/components/src/organisms/ShippingSimulation)

The `ShippingSimulation` component lets customers set the `PostalCode` and see the shipping options in the `PDP`.

<Tabs items={['Example', 'Code']} defaultIndex="0">
  <Tab>
    <OverviewSection>
      <ShippingSimulationUsage />
    </OverviewSection>
  </Tab>
  <Tab>
    ```tsx
    const shippingOptions = [
      {
          carrier: 'DHL',
          localizedEstimates: 'Up to 3 business days',
          price: 12,
      },
    ]

    const shippingLocation = "Alfredo Coutinho St"

    const [options, setOptions] = useState([])
    const [postalCode, setPostalCode] = useState("")
    const [location, setLocation] = useState("")
    const [error, setError] = useState("")
    const [displayClearButton, setDisplayClearButton] = useState(false)

    const handleSubmit = () => {
      if (isNaN(Number(postalCode)) || postalCode.length <= 0) {
          setError("Set a valid Postal Code")
          return
      }
      setOptions(shippingOptions)
      setLocation(shippingLocation)
      setDisplayClearButton(true)
    }

    const handleClear = () => {
        setOptions([])
        setLocation("")
        setDisplayClearButton(false)
        setPostalCode("")
        setError("")
    }

    const handleInput = (value: string) => {
        setError("")
        setPostalCode(value)
        setDisplayClearButton(false)
    }

    <ShippingSimulation
      onInput={(event) => handleInput(event.currentTarget.value)}
      onSubmit={handleSubmit}
      onClear={handleClear}
      location={location}
      options={options}
      displayClearButton={displayClearButton}
      formatter={(value) => value}
      errorMessage={error}
      postalCode={postalCode}
    />
    ```

  </Tab>
</Tabs>

## Usage

### Import the component

```tsx
import { ShippingSimulation } from "@faststore/ui";
```

### Import styles

To apply styles, include the following in your stylesheet:

```scss
@import "@faststore/ui/src/components/organisms/ShippingSimulation/styles.scss";
```

For details, see [Importing FastStore UI component styles](https://developers.vtex.com/docs/guides/faststore/using-themes-importing-ui-components-styles).
